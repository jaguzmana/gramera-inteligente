create schema pesoPlumaDb;

use pesoPlumaDb;

CREATE TABLE Ingredient (
    ingredient_id INT AUTO_INCREMENT PRIMARY KEY,
    ingredient_name VARCHAR(255) NOT NULL,
    ingredient_amount FLOAT(7,7),
    ingredient_unit VARCHAR(15) NOT NULL,
    ingredient_density FLOAT NOT NULL CHECK (ingredient_density >= 1e-6 AND ingredient_density <= 1e3)
);

CREATE TABLE Consume (
    consume_id INT AUTO_INCREMENT PRIMARY KEY,
    consume_date DATETIME NOT NULL,
    consume_amount FLOAT NOT NULL CHECK (consume_amount >= 1e-6 AND consume_amount <= 1e6),
    ingredient_id INT,
    FOREIGN KEY (ingredient_id) REFERENCES Ingredient(ingredient_id)
);

CREATE TABLE Recipe (
    recipe_id INT AUTO_INCREMENT PRIMARY KEY,
    recipe_name VARCHAR(255) NOT NULL
);

CREATE TABLE Amount (
    amount_id INT AUTO_INCREMENT PRIMARY KEY,
    amount_amount FLOAT NOT NULL CHECK (amount_amount >= 1e-6 AND amount_amount <= 1e6),
    amount_unit VARCHAR(15) NOT NULL,
    amount_description VARCHAR(255),
    ingredient_id INT,
    recipe_id INT,
    FOREIGN KEY (ingredient_id) REFERENCES Ingredient(ingredient_id),
    FOREIGN KEY (recipe_id) REFERENCES Recipe(recipe_id)
);
